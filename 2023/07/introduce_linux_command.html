    2. 常用命令

    命令 简介
    realpath filename 打印文件的绝对路径名
    free -h 查看总内存(RAM)及其使用情况
    df -hl 查看磁盘空间使用情况
    zcat filename.gz | head 在不解压缩文件的前提下, 查看文件部分内容, 输出到标准输出设备
    zless filename.gz 和上面命令的用途一致, 速度快, 推荐
    tar czf newfile.tar.gz oldfile 压缩文件
    tar xzf file.tar.gz 解压缩文件
    zip -r <newfile>.zip oldfile olddir 压缩文件
    unzip <filename>.zip 解压缩文件
    ps xu 查看当前用户的进程.更多信息
    kill -2 <pid> 默认行为就是退出程序
    pgrep <string> 根据正则表达式列出相应的进程
    pkill -2 <string> 根据正则表达式退出相应的进程.更多信息
    basename ~/a/b/c.txt 输出c.txt
    dirname ~/a/b/c.txt 输出/home/user/a/b
    rename "s/pattern_a/pattern_b/" *.txt 批量重命名所有的txt文件

    6. 查看目录的大小

    #查看当前目录的总大小
du -sh

#查看当前目录的各个子目录及文件的大小
du -sh *

du -h --max-depth=1 *
du -h --max-depth=1 ./
#两个命令结果不同, 后者只显示目录, 且显示隐藏目录

    7. Combined mkdir and cd

    1. function
    mkcdir ()
{
    mkdir -p -- "$1" &&
       cd -P -- "$1"
}

    Put the above code in the ~/.bashrc
    source ~/.bashrc to apply changes.
    After that simply run mkcdir foo or mkcdir "nested/path/in quotes".

    bash - Combined mkdir and cd - Unix & Linux Stack Exchange

    8. $_ and !$

    $ echo "$(date +%s)" # %s is seconds since epoch
1536984661
$ echo $_ !$ "$(date +%s)" # don't hit enter until a few seconds pass
1536984661 1536984665 1536984665

$echo a b c
a b c
$echo !:2
echo b
b

    9. 延续一个长命令: \

    把下面的命令保存的 test_1.sh, 运行, 不会有报错, 猜测加上 \ 后, 如果 \ 的下一行是空行, 则表明一个命令已经写完, 如果 \的下一行不是空行, 则会把下一行当做命令的一部分, 从而可能导致报错, 如 test_2.sh

    #test_1.sh
ls -a \
    -l \

hello() {
    echo "Hello world!" \

}
hello

#test_2.sh
hello() {
    echo "Hello world!" \
}
## test_2.sh: line 4: syntax error: unexpected end of file
# "}" 被当做第二行的一部分, 导致函数没有正常的 "}" 来结束

#test_3.sh
hello() {
        echo "Hello world!" \
}
}
hello
##output: Hello world! }

    我的PS1配置

    if [ $(hostname) == "admin" ]
then
    export PS1="\A \[\e[37;40m\]\[\e[32;40m\]\u\[\e[37;40m\]@\[\e[32;40m\]\h\[\e[33;40m\] | \$(grep -o 'AJ[A-Z]\{1,3\}[0-9]\{10\}-*[A-Z0-9]*' <<< \"\$PWD\") | \w | \W \[\e[37;40m\]\n\[\e[33;40m\]\$\[\e[0m\]"
else
    export PS1="\A \[\e[37;40m\]\[\e[32;40m\]\u\[\e[37;40m\]@\[\e[31;40m\]\h\[\e[33;40m\] | $(grep -o 'AJ[A-Z]\{1,3\}[0-9]\{10\}-*[A-Z0-9]*' <<< "$PWD") | \w | \W \[\e[37;40m\]\n\[\e[33;40m\]$\[\e[0m\]"
fi
