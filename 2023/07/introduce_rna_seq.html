    run_fastqc(){
 filedir=$1
 fqdir=$2
 ls ${filedir}*gz | while read id 
 do
  nohup fastqc $id -t 10 -o $fqdir &
 done
}

    5. 数据过滤

    run_trim_galore() {
 basedir=~/23-rna-seq-7-2/10-raw/
 outdir=~/23-rna-seq-7-2/21-trim-galore/
 for i in 08 09 12 13
 do
  fq1=SRR10395${i}_1.fastq.gz
  fq2=SRR10395${i}_2.fastq.gz
  path1=${basedir}${fq1}
  path2=${basedir}${fq2}
  #echo $path1 $path2 $outdir
  nohup trim_galore -q 28 --phred33 --length 30 --paired -e 0.1 --stringency 1 -o $outdir $path1 $path2 &
 done
}

run_trim_galore

    6. 序列比对

    • 运行 hisat2
    run_hisat2() {
 index=~/23-rna-seq-7-2/31-hisat2-index/hg19/genome
 indir=~/23-rna-seq-7-2/21-trim-galore/
 outdir=~/23-rna-seq-7-2/33-hisat2-result/
 #cat ~/apa/align/input |while read id
 prefix=SRR10395
 for id in 08 09 12 13
 do      #arr=( $id )
  val1=${indir}${prefix}${id}_1_val_1.fq.gz
  val2=${indir}${prefix}${id}_2_val_2.fq.gz
  outfile=${outdir}${prefix}${id}
  #echo -e "$val1\n$val2\n$outfile\n\n"
  nohup hisat2 -p 20 -x $index -1 $val1 -2 $val2 -S ${outfile}.sam  2> ${outfile}.hisat2.log &
 done
}

run_hisat2
    • 查看比对结果
    view_flagstat() {
 dir=~/23-rna-seq-7-2/33-hisat2-result/

 realpath ${dir}*.sam | while read id
 do
  #echo -e "$id\n${id}.flagstat.txt\n\n"
  nohup samtools view -@ 10 -S -b ${id}  | samtools flagstat - > ${id}.flagstat.txt &
 done
}

    7. 表达量计数

    • 将 sam 文件转化为 bam 文件
    sam_to_bam() {
 dir=~/23-rna-seq-7-2/33-hisat2-result/

 ls ${dir}*.sam | while read id
 do
  id=$(basename $id)
  #echo -e "$id\n${dir}${id}\n\n"
  nohup samtools view -bS ${dir}${id} > ${dir}../41-sam_to_bam/${id/.sam/.bam} &
 done
}

sam_to_bam
    • 对 bam 文件进行排序
    sort_bam() {
 dir=$1
 realpath ${dir}*.bam | while read id
 do
  filename=$(basename $id)
  output=${dir}../43-sorted-bam/${filename/.bam/.sort.bam}
  #echo $output
  nohup samtools sort -@ 10  $id -o $output &
 done
}

sort_bam ~/23-rna-seq-7-2/41-sam_to_bam/
    • 创建索引
    create_index() {
 dir=$1
 realpath ${dir}*.bam | while read id
 do
  #echo $id\n
  nohup samtools index -@ 10 $id &
 done
}

create_index ~/23-rna-seq-7-2/43-sorted-bam/
